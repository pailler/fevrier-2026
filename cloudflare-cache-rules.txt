# Règles de cache Cloudflare pour éviter les problèmes de chunks Next.js
# À appliquer dans Cloudflare Dashboard > Rules > Page Rules

# Règle 1: Ne pas mettre en cache les fichiers _next/static/
URL: iahome.fr/_next/static/*
Settings:
- Cache Level: Bypass
- Edge Cache TTL: 0 seconds

# Règle 2: Ne pas mettre en cache les fichiers de chunks
URL: iahome.fr/_next/static/chunks/*
Settings:
- Cache Level: Bypass
- Edge Cache TTL: 0 seconds

# Règle 3: Ne pas mettre en cache les fichiers CSS/JS de Next.js
URL: iahome.fr/_next/static/css/*
Settings:
- Cache Level: Bypass
- Edge Cache TTL: 0 seconds

URL: iahome.fr/_next/static/js/*
Settings:
- Cache Level: Bypass
- Edge Cache TTL: 0 seconds

# Règle 4: Cache normal pour le reste du site
URL: iahome.fr/*
Settings:
- Cache Level: Standard
- Edge Cache TTL: 1 hour
